(static) forEachDeep(arr, fn)

Source:
  • lightdash.esm.js, line 1320

Recursively iterates over each element in an array.

Example
const a = [2, 4, [1, 1, [16], 4]];

forEachDeep(a, (val, index, arr) => {
    arr[index] = index * val;
})
// a = [0, 4, [0, 1, [0], 12]]

(static) forEachEntry(obj, fn)

Source:

Iterates over each entry of an object.

Example
const a = {a: 1, b: 2};

forEachEntry(a, (key, val, index) => {
    a[key] = val * index;
})
// a = {a: 0, b: 2}
